#include <iostream>
#include <bits/stdc++.h>
#define int long long
using namespace std;


int32_t main()
{
    string s;
    cin>>s;
    int n=s.length();
    if(n==1){
        cout<<0<<endl;
        return 0;
    }
    if(n==2){
        cout<<1<<endl;
        return 0;
    }
    int isPal[n][n];
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)isPal[i][j]=0;
    }
    int R=n,C=0;
    while(R>0){
        int r = 0,c=C;
        while(r<R){
            if(r==c)isPal[r][c]=1;
            else if((r+1)==c){if(s[r]==s[c])isPal[r][c]=1;else isPal[r][c]=0;}
            else {
                if(s[r]==s[c] && isPal[r+1][c-1]==1){
                    isPal[r][c]=1;
                }
            }
            r++;c++;
        }
        R--;
        C++;
    }
    int sum1[n],sum2[n];
    for(int i=0;i<n;i++){
        int s1=0;
        for(int j=0;j<n;j++){
            s1+=isPal[i][j];
        }
        sum1[i]=s1;
    }
    for(int i=n-2;i>=0;i--)sum1[i]=sum1[i]+sum1[i+1];
    for(int j=0;j<n;j++){
        sum2[j]=0;
        for(int i=0;i<=j;i++){
            sum2[j]+=isPal[i][j];
        }
    }
    int ans=0;
    for(int i=n-2;i>=0;i--){
        ans+=(sum2[i]*sum1[i+1]);
    }
    cout<<ans<<endl;
   return 0;
}
