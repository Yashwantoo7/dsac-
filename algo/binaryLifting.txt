#include <iostream>
#include <bits/stdc++.h>
using namespace std;

//find a node k level above the queried node

void binaryLifting(int src,int par,int up[][20],vector<int>tree[]){
    up[src][0]=par;
    for(int i=1;i<20;i++){
        if(up[src][i-1]!=-1)
            up[src][i]=up[up[src][i-1]][i-1];
        else up[src][i]=-1;
    }
    
    for(int child:tree[src]){
        if(child!=par){
            binaryLifting(child,src,up,tree);
        }
    }
}

int query(int node,int k,vector<int>tree[],int up[][20]){
    
    if(node==-1||k==0){
        return node;
    }
    for(int i=19;i>=0;i--){
        if(k>=(1<<i)){
            return query(up[node][i],k-(1<<i),tree,up);
        }
    }
}

void solve(){
    int n,q;
    cin>>n>>q;
    vector<int>tree[n+1];
    int up[2001][20];
    for(int i=1;i<n;i++){
        int u,v;
        cin>>u>>v;
        tree[u].push_back(v);
        tree[v].push_back(u);
    }
    binaryLifting(1,-1,up,tree);
    while(q--){
        int node,k;
        cin>>node>>k;
        cout<<query(node,k,tree,up)<<endl;
    }
    
}


int main()
{
   int t;
   cin>>t;
   while(t--){
       solve();
   }
   return 0;
}
