#include <iostream>
#include <bits/stdc++.h>

using namespace std;

bool poss(double x,int arr[],int n,int k){
    double temp[n];
    for(int i=0;i<n;i++){
        temp[i] = arr[i] - x;
    }
    double sum = 0;
    for(int i=0;i<k;i++){
        sum+=temp[i];
    }
    if(sum>=0)return true;
    double last = 0;
    for(int i=0,j=k;j<n;j++,i++){
        last += temp[i];
        sum += temp[j];
        if(last<0){
            sum -= last;
            last = 0;
        }
        if(sum>=0)return true;
    }
    return false;
}

int main()
{
   int n,k;
   cin>>n>>k; 
   
   int arr[n];
   for(int i=0;i<n;i++)cin>>arr[i];
   
   double ans = INT_MIN;
   double l = INT_MIN;
   double r = INT_MAX;
   
   while(r-l>0.00005){
       double mid =l +  (r-l)/2.0;
       if(poss(mid,arr,n,k)){
           l = mid;
           ans = mid;
       }
       else{
           r = mid;
       }
   }
   cout<<ans<<endl;
   return 0;
}
