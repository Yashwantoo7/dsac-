#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int start[100];
int endd[100];
vector<int>dfs_order;

void dfs(int u,int &count, vector<int>tree[]){
    start[u]=count;
    count++;
    dfs_order.push_back(u);
    for(auto v:tree[u]){
        dfs(v,count,tree);
    }
    endd[u]=count;
}

void Print(int n){
    for(int i=0;i<n;i++){
        for(int j=start[i];j<endd[i];j++){
            cout<<dfs_order[j]<<" ";
        }
        cout<<endl;
    }
}

void solve(){
    int n;
    cin>>n;
    vector<int>tree[n];
    for(int i=0;i<n-1;i++){
        int u,v;
        cin>>u>>v;
        tree[u].push_back(v);
    }
    int count=0;
    dfs(0,count,tree);
    Print(n);
}

int main()
{
   solve();
   return 0;
}
